                              # 安卓环境

                              # 下载android studio并安装jdk1.8+

                              # mac下java环境变量配置
vi ~/.zshrc
### 将内容添加
JAVA_HOME="/Library/Java/JavaVirtualMachines/jdk1.8.0_181.jdk/Contents/Home"
export JAVA_HOME
CLASS_PATH="$JAVA_HOME/lib"
PATH=“$PATH:$JAVA_HOME/bin”
###
# 重启配置
source ~/.zshrc
java -version
